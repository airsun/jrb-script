port vlan ${vlan} 0/0 1

interface vlanif ${vlan}

ip address ${ip} 255.255.255.0

quit

ip route-static 211.103.1.22 255.255.255.255 ${gatway}

voip

ip address media ${ip} ${gatway}

ip address signaling ${ip}

qos vlan media ${ip} priority 4

qos vlan signaling ${ip} priority 4

quit

interface h248 0
y

digitmap-timer long 5

digitmap-timer short 3

if-h248 attribute mgip ${ip} mgport 2944 transfer udp

if-h248 attribute primary-mgc-ip1 211.103.1.22 primary-mgc-port 2944 mgc-domain-name1 ${domain1}

if-h248 attribute mg-media-ip1 ${ip} MIDtype domainName

if-h248 attribute profile-negotiation enable start-negotiate-version 1

if-h248 attribute domainName ${domain1}

tid-format rtp prefix RTP/ template 32

mg-software parameter 2 2

quit

mgpstnuser batadd 0/${board_start}/0 0/${board_end}/63 0 terminalid 0 step 1

quit

pstnport

quit
